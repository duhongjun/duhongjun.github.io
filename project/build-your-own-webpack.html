      # 前言

      本文内容是参考了 Ronen Amiel - Build Your Own Webpack， 能翻墙的可以去看一下原视频。 webpack本身很复杂， 这次只模拟实现一下如何分析es6的模块依赖及如何将ES6代码转为ES5。

      # 准备过程

      由于需要借助babel，所以需要安装以下插件：

      babylon， @babel/core，@babel/parser，@babel/traverse，@babel/preset-env

      以及我们用来测试打包的文件：

      // entry.js
import message from './message'
console.log(message)

// message.js
import { name } from './name.js'
export default `hello ${name}!`

// name.js
export const name = 'world';

      以及我们的打包器 bundler.js, 目录结构如下：

      - example
    - entry.js
    - message.js
    - name.js
- bundler.js

      # 开始动手实现

      # 分析依赖思路

      我们使用webpack的时候都会指定入口文件，所以我们就从入口文件开始分析，读取入口文件的内容，然后将其转化为AST，从AST中读取依赖相关的信息，我们将这个函数命名为createAsset，初步代码如下：

      const fs = require('fs')
const babylon = require('babylon')
const traverse = require('babel-traverse').default

function createAssets(filename) {
  // 获取文件内容
  const content = fs.readFileSync(filename, 'utf-8')
  // 将文件内容转换为AST
  const ast = babylon.parse(content, {
    sourceType: 'module',
  })
  // 遍历AST
  traverse(ast, {
    //找到import语法的对应节点
    ImportDeclaration: ({ node }) => {
      console.log(node)
    },
  })
}

createAssets('./example/entry.js')

      我们看一下Node节点是什么样的： 可以看到node.source.value就是我们想要的依赖模块路径。 所以我们可以创建一个依赖数组， 用于存储模块路径，然后返回当前模块的信息，完善后的代码如下：

      const fs = require('fs')
const babylon = require('babylon')
const traverse = require('babel-traverse').default

let ID = 0

function createAssets(filename) {
  // 获取文件内容
  const content = fs.readFileSync(filename, 'utf-8')
  // 将文件内容转换为AST
  const ast = babylon.parse(content, {
    sourceType: 'module',
  })
  const dependencies = []
  // 遍历AST
  traverse(ast, {
    //找到import语法的对应节点
    ImportDeclaration: ({ node }) => {
      dependencies.push(node.source.value)
    },
  })
  const id = ID++
  return {
    id,
    filename,
    dependencies,
  }
}

const mainAssets = createAssets('./example/entry.js')
console.log(mainAssets)
